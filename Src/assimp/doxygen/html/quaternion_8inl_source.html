85 // Constructs a quaternion from a rotation matrix
86 template<typename TReal>
87 inline aiQuaterniont<TReal>::aiQuaterniont( const aiMatrix3x3t<TReal> &pRotMatrix)
88 {
89  TReal t = pRotMatrix.a1 + pRotMatrix.b2 + pRotMatrix.c3;
90 
91  // large enough
92  if( t > static_cast<TReal>(0))
93  {
94  TReal s = std::sqrt(1 + t) * static_cast<TReal>(2.0);
95  x = (pRotMatrix.c2 - pRotMatrix.b3) / s;
96  y = (pRotMatrix.a3 - pRotMatrix.c1) / s;
97  z = (pRotMatrix.b1 - pRotMatrix.a2) / s;
98  w = static_cast<TReal>(0.25) * s;
99  } // else we have to check several cases
100  else if( pRotMatrix.a1 > pRotMatrix.b2 && pRotMatrix.a1 > pRotMatrix.c3 )
101  {
102  // Column 0:
103  TReal s = std::sqrt( static_cast<TReal>(1.0) + pRotMatrix.a1 - pRotMatrix.b2 - pRotMatrix.c3) * static_cast<TReal>(2.0);
104  x = static_cast<TReal>(0.25) * s;
105  y = (pRotMatrix.b1 + pRotMatrix.a2) / s;
106  z = (pRotMatrix.a3 + pRotMatrix.c1) / s;
107  w = (pRotMatrix.c2 - pRotMatrix.b3) / s;
108  }
109  else if( pRotMatrix.b2 > pRotMatrix.c3)
110  {
111  // Column 1:
112  TReal s = std::sqrt( static_cast<TReal>(1.0) + pRotMatrix.b2 - pRotMatrix.a1 - pRotMatrix.c3) * static_cast<TReal>(2.0);
113  x = (pRotMatrix.b1 + pRotMatrix.a2) / s;
114  y = static_cast<TReal>(0.25) * s;
115  z = (pRotMatrix.c2 + pRotMatrix.b3) / s;
116  w = (pRotMatrix.a3 - pRotMatrix.c1) / s;
117  } else
118  {
119  // Column 2:
120  TReal s = std::sqrt( static_cast<TReal>(1.0) + pRotMatrix.c3 - pRotMatrix.a1 - pRotMatrix.b2) * static_cast<TReal>(2.0);
121  x = (pRotMatrix.a3 + pRotMatrix.c1) / s;
122  y = (pRotMatrix.c2 + pRotMatrix.b3) / s;
123  z = static_cast<TReal>(0.25) * s;
124  w = (pRotMatrix.b1 - pRotMatrix.a2) / s;
125  }
126 }

129 // Construction from euler angles
130 template<typename TReal>
131 inline aiQuaterniont<TReal>::aiQuaterniont( TReal fPitch, TReal fYaw, TReal fRoll )
132 {
133  const TReal fSinPitch(std::sin(fPitch*static_cast<TReal>(0.5)));
134  const TReal fCosPitch(std::cos(fPitch*static_cast<TReal>(0.5)));
135  const TReal fSinYaw(std::sin(fYaw*static_cast<TReal>(0.5)));
136  const TReal fCosYaw(std::cos(fYaw*static_cast<TReal>(0.5)));
137  const TReal fSinRoll(std::sin(fRoll*static_cast<TReal>(0.5)));
138  const TReal fCosRoll(std::cos(fRoll*static_cast<TReal>(0.5)));
139  const TReal fCosPitchCosYaw(fCosPitch*fCosYaw);
140  const TReal fSinPitchSinYaw(fSinPitch*fSinYaw);
141  x = fSinRoll * fCosPitchCosYaw - fCosRoll * fSinPitchSinYaw;
142  y = fCosRoll * fSinPitch * fCosYaw + fSinRoll * fCosPitch * fSinYaw;
143  z = fCosRoll * fCosPitch * fSinYaw - fSinRoll * fSinPitch * fCosYaw;
144  w = fCosRoll * fCosPitchCosYaw + fSinRoll * fSinPitchSinYaw;
145 }

147 // ---------------------------------------------------------------------------
148 // Returns a matrix representation of the quaternion
149 template<typename TReal>
150 inline aiMatrix3x3t<TReal> aiQuaterniont<TReal>::GetMatrix() const
151 {
152  aiMatrix3x3t<TReal> resMatrix;
153  resMatrix.a1 = static_cast<TReal>(1.0) - static_cast<TReal>(2.0) * (y * y + z * z);
154  resMatrix.a2 = static_cast<TReal>(2.0) * (x * y - z * w);
155  resMatrix.a3 = static_cast<TReal>(2.0) * (x * z + y * w);
156  resMatrix.b1 = static_cast<TReal>(2.0) * (x * y + z * w);
157  resMatrix.b2 = static_cast<TReal>(1.0) - static_cast<TReal>(2.0) * (x * x + z * z);
158  resMatrix.b3 = static_cast<TReal>(2.0) * (y * z - x * w);
159  resMatrix.c1 = static_cast<TReal>(2.0) * (x * z - y * w);
160  resMatrix.c2 = static_cast<TReal>(2.0) * (y * z + x * w);
161  resMatrix.c3 = static_cast<TReal>(1.0) - static_cast<TReal>(2.0) * (x * x + y * y);
162 
163  return resMatrix;
164 }

167 // Construction from an axis-angle pair
168 template<typename TReal>
169 inline aiQuaterniont<TReal>::aiQuaterniont( aiVector3t<TReal> axis, TReal angle)
170 {
171  axis.Normalize();
172 
173  const TReal sin_a = std::sin( angle / 2 );
174  const TReal cos_a = std::cos( angle / 2 );
175  x = axis.x * sin_a;
176  y = axis.y * sin_a;
177  z = axis.z * sin_a;
178  w = cos_a;
179 }
180 // ---------------------------------------------------------------------------
181 // Construction from am existing, normalized quaternion
182 template<typename TReal>
183 inline aiQuaterniont<TReal>::aiQuaterniont( aiVector3t<TReal> normalized)
184 {
185  x = normalized.x;
186  y = normalized.y;
187  z = normalized.z;
188 
189  const TReal t = static_cast<TReal>(1.0) - (x*x) - (y*y) - (z*z);
190 
191  if (t < static_cast<TReal>(0.0)) {
192  w = static_cast<TReal>(0.0);
193  }
194  else w = std::sqrt (t);
195 }

197 // ---------------------------------------------------------------------------
198 // Performs a spherical interpolation between two quaternions
199 // Implementation adopted from the gmtl project. All others I found on the net fail in some cases.
200 // Congrats, gmtl!
201 template<typename TReal>
202 inline void aiQuaterniont<TReal>::Interpolate( aiQuaterniont& pOut, const aiQuaterniont& pStart, const aiQuaterniont& pEnd, TReal pFactor)
203 {
204  // calc cosine theta
205  TReal cosom = pStart.x * pEnd.x + pStart.y * pEnd.y + pStart.z * pEnd.z + pStart.w * pEnd.w;
206 
207  // adjust signs (if necessary)
208  aiQuaterniont end = pEnd;
209  if( cosom < static_cast<TReal>(0.0))
210  {
211  cosom = -cosom;
212  end.x = -end.x; // Reverse all signs
213  end.y = -end.y;
214  end.z = -end.z;
215  end.w = -end.w;
216  }
217 
218  // Calculate coefficients
219  TReal sclp, sclq;
220  if( (static_cast<TReal>(1.0) - cosom) > static_cast<TReal>(0.0001)) // 0.0001 -> some epsillon
221  {
222  // Standard case (slerp)
223  TReal omega, sinom;
224  omega = std::acos( cosom); // extract theta from dot product's cos theta
225  sinom = std::sin( omega);
226  sclp = std::sin( (static_cast<TReal>(1.0) - pFactor) * omega) / sinom;
227  sclq = std::sin( pFactor * omega) / sinom;
228  } else
229  {
230  // Very close, do linear interp (because it's faster)
231  sclp = static_cast<TReal>(1.0) - pFactor;
232  sclq = pFactor;
233  }
234 
235  pOut.x = sclp * pStart.x + sclq * end.x;
236  pOut.y = sclp * pStart.y + sclq * end.y;
237  pOut.z = sclp * pStart.z + sclq * end.z;
238  pOut.w = sclp * pStart.w + sclq * end.w;
239 }

242 template<typename TReal>
243 inline aiQuaterniont<TReal>& aiQuaterniont<TReal>::Normalize()
244 {
245  // compute the magnitude and divide through it
246  const TReal mag = std::sqrt(x*x + y*y + z*z + w*w);
247  if (mag)
248  {
249  const TReal invMag = static_cast<TReal>(1.0)/mag;
250  x *= invMag;
251  y *= invMag;
252  z *= invMag;
253  w *= invMag;
254  }
255  return *this;
256 }

261 {

279 template<typename TReal>
280 inline aiVector3t<TReal> aiQuaterniont<TReal>::Rotate (const aiVector3t<TReal>& v)
281 {
282  aiQuaterniont q2(0.f,v.x,v.y,v.z), q = *this, qinv = q;
283  qinv.Conjugate();
284 
285  q = q*q2*qinv;
286  return aiVector3t<TReal>(q.x,q.y,q.z);
